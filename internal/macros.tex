		\huge{\textbf{\uppercase{#1}}}
	\end{center}\vspace{-1mm}\par
}

\newcommand{\comentry}[2]{%
	\begin{itemize}[
		align=left,
		leftmargin=2em,
  		itemindent=0pt,
  		labelsep=0pt,
 		labelwidth=2em]
	\item[{#1}] #2
	\end{itemize}
}
